/*Exercício 4: construir um programa em Java que leia um número inteiro e imprima
“PAR”, se onúmero for par e imprima “ÍMPAR”, se o número for ímpar.*/

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Digite o valor: ");
        int val = sc.nextInt();

        if(val % 2 == 0 || val == 0){
            System.out.print("O dígito é par");
        }
        else {
            System.out.print("O dígito é ímpar");
        }
        sc.close();
    }
}
