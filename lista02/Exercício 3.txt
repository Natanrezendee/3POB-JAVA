/*Exercício 3: construir um programa em Java que leia 2 números inteiros e imprima o maior
número lido.*/

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Digite o primeiro valor: ");
        int val1 = sc.nextInt();
        System.out.print("Digite o segundo valor: ");
        int val2 = sc.nextInt();

        if(val1 > val2){
            System.out.println("O dígito " + val1 + " (primeiro elemento) é o maior");

        }
        else if(val2 > val1){
            System.out.println("O dígito " + val2 + " (segundo elemento) é o maior");
        }
        else {
            System.out.print("Ambos os valores são iguais");
        }
        sc.close();
    }
}
