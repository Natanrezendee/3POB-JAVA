/*Exercício 12: construir um programa em Java que leia os valores de A, B e C e que imprima
“Não forma triângulo”, se um dos valores for maior que a soma dos outros 2, caso contrário,
imprimir qual é o tipo do triângulo, que pode ser “Equilátero”, se tiver os 3 lados iguais,
“Isósceles”, se tiver 2 lados iguais e “Escaleno”, se os 3 lados forem diferentes.*/

import java.util.Scanner;

 public class Main {
     public static void main(String[] args) {
         Scanner sc = new Scanner(System.in);

         System.out.print("Insira o valor do primeiro lado ");
         float ladoA = sc.nextFloat();
         System.out.print("Insira o valor do primeiro lado ");
         float ladoB = sc.nextFloat();
         System.out.print("Insira o valor do primeiro lado ");
         float ladoC = sc.nextFloat();

         if(ladoA > ladoB + ladoC){
             System.out.print("Não é um triângulo!");
         }
         else if(ladoB > ladoA + ladoC){
             System.out.print("Não é um triângulo!");
         }
         else if(ladoC > ladoA + ladoB){
             System.out.print("Não é um triângulo!");
         }
         else{
             if(ladoA == ladoB && ladoA == ladoC){
                 System.out.print("É um triângulo Equilátero!");
             }
            else if(ladoA != ladoB && ladoA != ladoC && ladoB != ladoC){
                 System.out.print("É um triângulo escaleno!");
             }
            else{
                 System.out.print("É um triângulo Isósceles!");
             }
         }
        sc.close();
 }}
